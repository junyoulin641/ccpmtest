**KPI Calculations:**

```python
def update_kpis(self, df):
    # Row count
    row_count = len(df)

    # Column count
    col_count = len(df.columns)

    # Memory usage (approximate)
    memory_mb = df.memory_usage(deep=True).sum() / (1024 ** 2)

    # Missing values count
    missing_count = df.isnull().sum().sum()

    # Update KPI cards
    self.rows_card.update_value(f"{row_count:,}")
    self.cols_card.update_value(col_count)
    self.memory_card.update_value(f"{memory_mb:.2f} MB")
    self.missing_card.update_value(missing_count)
```

**Summary Statistics Table:**

Use QTableWidget to display stats:
```python
def update_statistics_table(self, df):
    numeric_cols = df.select_dtypes(include=[np.number]).columns
    stats_df = df[numeric_cols].describe()

    self.stats_table.setRowCount(len(stats_df))
    self.stats_table.setColumnCount(len(numeric_cols))
    self.stats_table.setHorizontalHeaderLabels(numeric_cols.tolist())
    self.stats_table.setVerticalHeaderLabels(stats_df.index.tolist())

    for i, row in enumerate(stats_df.index):
        for j, col in enumerate(numeric_cols):
            value = stats_df.loc[row, col]
            item = QTableWidgetItem(f"{value:.2f}")
            self.stats_table.setItem(i, j, item)
```

def update_data_preview(self, df):

def create_datatypes_chart(self, df):
    type_counts = df.dtypes.value_counts()

    fig, ax = plt.subplots(figsize=(4, 4))
    ax.pie(type_counts.values, labels=type_counts.index.astype(str),
           autopct='%1.0f%%', startangle=90)
    ax.set_title("Data Types Distribution")

    # Embed in widget
    canvas = FigureCanvas(fig)
    return canvas

**Missing Values Indicator:**

Show which columns have missing values:
```python
def show_missing_values(self, df):
    missing = df.isnull().sum()
    missing_cols = missing[missing > 0]

    if len(missing_cols) > 0:
        msg = "Columns with missing values:\n"
        for col, count in missing_cols.items():
            pct = (count / len(df)) * 100
            msg += f"- {col}: {count} ({pct:.1f}%)\n"
        self.missing_label.setText(msg)
    else:
        self.missing_label.setText("No missing values")
```
